\end{itemize}
\end{block}

	\begin{itemize}
	\item<+-> Addition is hard for EVDD~\cite{fargier2013semiring}, hence Hadamard gates are hard for QMDD
	\item<+-> QMDD is prone to numerical instability~\cite{niemann2020overcoming}

	\end{itemize}
